            <section id="intro">
                <h1>MaazDB v11.6 Documentation</h1>
                <p>MaazDB v11.6 is an ultra-high-performance, columnar, disk-based relational database engine written in Rust. It utilizes <strong>LSM Trees (Log-Structured Merge-Trees)</strong> for fast writes, <strong>Columnar Storage</strong> for analytical reads, and <strong>XOR Encryption</strong> for data security.</p>
            </section>

            <section id="install">
                <h2>1. Installation & Configuration</h2>
                <h3>Configuration File (`maazdb.toml`)</h3>
                <p>The configuration file is generated automatically on first launch if it does not exist.</p>
                <ul>
                    <li><strong>Linux:</strong> <code>/etc/maazdb/maazdb.toml</code></li>
                    <li><strong>Windows:</strong> <code>C:\Program Files\MaazDB\maazdb.toml</code></li>
                </ul>
                <pre><code>[server]
host = "0.0.0.0"
port = 8888

[storage]
data_dir = "data"</code></pre>
            </section>

            <section id="auth">
                <h2>2. Authentication</h2>
                <p>Security is mandatory. You must login before running queries.</p>
                <pre><code>LOGIN admin admin;</code></pre>
                
                <h3>Manage Users</h3>
                <pre><code>CREATE USER maaz WITH PASSWORD 'secure';
DROP USER maaz;</code></pre>
            </section>

            <section id="db-mgmt">
                <h2>3. Database Management</h2>
                <p>MaazDB is a multi-tenant system. You must Create and Use a database.</p>
                <pre><code>CREATE DATABASE inventory_system;
USE inventory_system;
BACKUP snapshot_v1; -- Creates a backup folder</code></pre>
            </section>

            <section id="types">
                <h2>4. Data Types</h2>
                <p>MaazDB uses Strict Static Typing.</p>
                <table>
                    <thead>
                        <tr><th>Type</th><th>Storage</th><th>Description</th></tr>
                    </thead>
                    <tbody>
                        <tr><td>SERIAL</td><td>Int32</td><td>Auto-increment Primary Key</td></tr>
                        <tr><td>TEXT</td><td>UTF-8</td><td>String values</td></tr>
                        <tr><td>INT</td><td>Int32</td><td>Whole numbers</td></tr>
                        <tr><td>DOUBLE</td><td>Float64</td><td>Decimal numbers (Prices)</td></tr>
                        <tr><td>BOOL</td><td>Byte</td><td>true/false</td></tr>
                        <tr><td>TIMESTAMP</td><td>ISO-8601</td><td>Date and Time</td></tr>
                    </tbody>
                </table>
            </section>

            <section id="ddl">
                <h2>5. DDL - Create Table</h2>
                <pre><code>CREATE TABLE products (
    pid SERIAL PRIMARY,
    title TEXT,
    price DOUBLE,
    in_stock BOOL
);

DROP TABLE products;</code></pre>
            </section>

            <section id="dml-insert">
                <h2>6. INSERT Data</h2>
                <p>Add new rows to a table. You can specify columns explicitly.</p>
                <pre><code>-- Explicit Insert
INSERT INTO products (title, price, in_stock) VALUES ('Gaming Mouse', 49.99, true);

-- Auto-Defaults (ID generates automatically)
INSERT INTO products (title, price) VALUES ('Keyboard', 89.99);</code></pre>
            </section>

            <section id="dml-select">
                <h2>7. SELECT Data</h2>
                <p>Retrieve data with powerful filtering and sorting.</p>
                
                <h3>Basic Select</h3>
                <pre><code>SELECT products;</code></pre>

                <h3>Filtering (WHERE)</h3>
                <pre><code>-- Find items cheaper than $50
SELECT products WHERE price < 50.0;

-- Find items that are in stock AND expensive
SELECT products WHERE in_stock = true AND price > 100.0;</code></pre>

                <h3>Sorting (ORDER BY)</h3>
                <pre><code>-- Sort by Price (High to Low)
SELECT products ORDER BY price DESC;

-- Sort by Title (A-Z)
SELECT products ORDER BY title ASC;</code></pre>

                <h3>Limiting Results</h3>
                <pre><code>-- Get top 5 most expensive items
SELECT products ORDER BY price DESC LIMIT 5;</code></pre>
            </section>

            <section id="dml-update">
                <h2>8. UPDATE Data</h2>
                <p>Modify existing records based on a condition.</p>
                <pre><code>-- Update price for a specific item
UPDATE products SET price = 45.00 WHERE title = 'Gaming Mouse';

-- Mark all items as out of stock (Dangerous!)
UPDATE products SET in_stock = false WHERE price < 10.0;</code></pre>
            </section>

            <section id="dml-delete">
                <h2>9. DELETE Data</h2>
                <p>Remove records permanently.</p>
                <pre><code>-- Delete a specific item by ID
DELETE FROM products WHERE pid = 5;

-- Delete all out of stock items
DELETE FROM products WHERE in_stock = false;</code></pre>
            </section>

document.addEventListener("DOMContentLoaded", () => {
    // 1. Detect Operating System
    const osMessage = document.getElementById('os-message');
    const platform = navigator.platform.toLowerCase();
    let os = "Unknown";

    if (platform.includes('win')) os = "Windows";
    else if (platform.includes('linux')) os = "Linux";
    else if (platform.includes('mac')) os = "macOS";

    // 2. Highlight and display message
    if(osMessage && os !== "Unknown") {
        osMessage.innerHTML = `<i class="fas fa-check-circle"></i> Detected OS: ${os} - Recommended Download Highlighted below.`;
        
        if (os === "Windows") {
            document.querySelector('.dl-card.windows').style.borderColor = '#dc2626';
            document.querySelector('.dl-card.windows').style.background = '#1e293b';
        } else if (os === "Linux") {
            document.querySelector('.dl-card.linux').style.borderColor = '#dc2626';
             document.querySelector('.dl-card.linux').style.background = '#1e293b';
        } else if (os === "macOS") {
            document.querySelector('.dl-card.mac').style.borderColor = '#dc2626';
             document.querySelector('.dl-card.mac').style.background = '#1e293b';
        }
    }
});
